### 方法 3: 手动检查步骤

#### 步骤 1: 检查登录状态

在控制台运行:
```javascript
console.log('Token:', localStorage.getItem('token'));
console.log('Username:', localStorage.getItem('username'));
```

**预期结果:**
- Token: 存在 (长字符串)
- Username: 存在 (你的用户名)

**如果没有数据:**
```javascript
// 重新登录
localStorage.setItem('username', 'admin');
// 或使用测试登录
testLogin("admin", "admin123");
```

#### 步骤 2: 检查 DOM 元素

在控制台运行:
```javascript
const el = document.getElementById('currentUser');
console.log('Element exists:', !!el);
console.log('Current text:', el ? el.textContent : 'N/A');
```

**预期结果:**
- Element exists: true
- Current text: "用户: admin" (或其他用户名)

**如果元素不存在:**
- 检查 HTML 结构是否有 `<span id="currentUser">`
- 检查是否有 JavaScript 错误阻止页面加载

#### 步骤 3: 手动修复显示

在控制台运行:
```javascript
const username = localStorage.getItem('username');
const el = document.getElementById('currentUser');
if (el && username) {
    el.textContent = '用户: ' + username;
    console.log('✅ 已修复:', el.textContent);
} else {
    console.log('❌ 修复失败');
}
```

## 🛠️ 常见问题和解决方案

### 问题 1: 用户名一直显示 "未知"

**原因:**
- localStorage 中没有存储 username
- 登录时没有正确保存用户信息

**解决方案:**
```javascript
// 方法 1: 手动设置
localStorage.setItem('username', 'admin');
location.reload();

// 方法 2: 重新登录
// 在登录页面重新登录
```

### 问题 2: 控制台显示 "currentUser 元素不存在"

**原因:**
- HTML 结构缺少 `<span id="currentUser">`
- 脚本执行时机过早，DOM 还未加载

**解决方案:**
1. 检查 HTML 中是否有 `<span id="currentUser">`
2. 确保脚本在 DOM 加载后执行
3. 检查是否有 JavaScript 错误

### 问题 3: 登录后立即跳转回登录页

**原因:**
- Token 无效或过期
- Token 验证失败
- API 认证问题

**解决方案:**
```javascript
// 检查 token
const token = localStorage.getItem('token');
console.log('Token:', token);

// 清除并重新登录
localStorage.clear();
// 重新登录
```

### 问题 4: 浏览器控制台有错误

**常见错误:**

1. **"Cannot read property 'textContent' of null"**
   - DOM 元素不存在
   - 检查元素 ID 是否正确

2. **"window.shared is undefined"**
   - shared.js 未加载
   - 检查脚本加载顺序

3. **"Failed to fetch"**
   - API 请求失败
   - 检查后端服务是否运行

### 4. API 检查

```javascript
// 测试 API 连接
async function checkAPI() {
    const token = localStorage.getItem('token');

    if (!token) {
        console.log('❌ 没有 token，请先登录');
        return;
    }

    try {
        // 测试用户 API
        const userRes = await fetch('/api/v1/users/me', {
            headers: {
                'Authorization': `Bearer ${token}`,
                'Content-Type': 'application/json'
            }
        });

        if (userRes.ok) {
            const user = await userRes.json();
            console.log('✅ 用户 API 正常:', user.username);
        } else {
            console.log('❌ 用户 API 失败:', userRes.status);
        }

        // 测试通信机 API
        const commRes = await fetch('/api/v1/communications', {
            headers: {
                'Authorization': `Bearer ${token}`,
                'Content-Type': 'application/json'
            }
        });

        if (commRes.ok) {
            const comms = await commRes.json();
            console.log('✅ 通信机 API 正常，数量:', comms.length);
        } else {
            console.log('❌ 通信机 API 失败:', commRes.status);
        }

    } catch (error) {
        console.log('❌ API 异常:', error.message);
    }
}

checkAPI();
```

## 🚀 快速修复命令

复制以下命令到控制台，一次性执行所有检查和修复:

```javascript
(async function() {
    console.log('=== 开始快速修复 ===');

    // 1. 检查登录状态
    const token = localStorage.getItem('token');
    const username = localStorage.getItem('username');

    if (!token || !username) {
        console.log('❌ 未登录，尝试测试登录...');
        const formData = new URLSearchParams();
        formData.append('username', 'admin');
        formData.append('password', 'admin123');

        try {
            const res = await fetch('/api/v1/users/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: formData
            });

            if (res.ok) {
                const data = await res.json();
                localStorage.setItem('token', data.access_token);
                localStorage.setItem('username', 'admin');
                console.log('✅ 登录成功');
            } else {
                console.log('❌ 登录失败');
                return;
            }
        } catch (error) {
            console.log('❌ 登录异常:', error.message);
            return;
        }
    }

    // 2. 更新用户显示
    const currentUserEl = document.getElementById('currentUser');
    if (currentUserEl) {
        currentUserEl.textContent = '用户: ' + localStorage.getItem('username');
        console.log('✅ 用户显示已更新:', currentUserEl.textContent);
    } else {
        console.log('❌ currentUser 元素不存在');
    }

    // 3. 重新加载数据
    if (window.communications && window.communications.loadCommunications) {
        console.log('⏳ 重新加载通信机列表...');
        await window.communications.loadCommunications();
        console.log('✅ 通信机列表已更新');
    }

    console.log('=== 修复完成 ===');
})();
```

   - 排除浏览器兼容性问题

   - Cmd + Shift + Delete (Mac)
   - 清除缓存和 Cookie

4. **检查服务器状态**
   - 后端服务是否正常运行
   - 数据库连接是否正常
   - API 端点是否可访问
